DOM Fiddler v1.0 by Ammar Haider

A tool to generate raw HTML code to manipulate the DOM. To be passed into a global container such as an Adobe Target global mbox. Uses CSS and jQuery.
<!-- 1. USE CSS TO HIDE/REMOVE THE ELEMENT -->
<style type= "text/css">
#banner img{visibility:hidden;}
</style>
<!-- 2. MANIPULATE THE ELEMENT WITH JAVASCRIPT/JQUERY -->
<script>
// MUST WAIT FOR THE DOCUMENT TO LOAD FIRST
$(function() {
// THIS EXAMPLE SHOWS THE SRC ATTRIBUTE OF AN IMAGE BEING CHANGED
$("#banner img").attr("src","http://www.example.co.uk/images/new-image.jpg");
// 3. SHOW THE ELEMENT WITH JAVASCRIPT/JQUERY
$('#banner img').css('visibility', 'visible');
});
</script>
